program QuickSort;

classes
	QuickSortClass
	methods
		method quickSort(ArrayIntenger: vetor, Intenger: inicio, fim)

variables
	QuickSortClass : quick_sort;
	ArrayInteger : vetor = [4,8,5,9,7];

methods
	QuickSortClass.quickSort
	vars
		ArrayIntenger: vetor;
		Intenger: inicio;
		Intenger: fim 
		Intenger: i,j,pivo,aux
	begin 
		i := inicio;
		j := fim-1;
		pivo := vetor[(inicio+fim)/2]
		while i <= j do
			while vetor[i] < pivo and i < fim do 
				i++
			end
			
			while vetor[j] > pivo and j > inicio do
				j--
			end

			if i <= j then
				aux := vetor[i];
				vetor[i] := vetor[j];
				vetor[j] := aux;
				i++;
				j--
			end

		end	 
		
		if j > inicio then 
			quick_sort.quickSort(vetor,inicio, j+1)
		end

		if i < fim then
			quick_sort.quickSort(vetor,i,fim)
		end
	end

begin
	quick_sort.quickSort(vetor,0,vetor.size())
end.
